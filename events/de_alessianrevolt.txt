		}
		has_country_modifier = cymmerian_invasion_modifier
	}
	
	option = {
		name = alessian_revolt.5.a
		if = {
			limit = {
				NOT = {
					capital_scope = {
						province_id = 1288
					}
				}
			}
			set_capital = 1288
			hidden_effect = {
				p:1288.state = {
					set_state_capital = 1288
				}
			}
		}
		p:1288 = {
			capital_formable_medium_effect = yes
			formable_capital_modifier_normal_effect = yes
		}
		add_stability = 25
	}
}
