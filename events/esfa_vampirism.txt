namespace = vampirism

vampirism.10 = {
	type = country_event
	hidden = yes
	
	trigger = {
		current_ruler = {
			has_trait = full_vampire
		}
		religion = night_pantheon
	}
	
	immediate = { 
		random_character = {
			limit = {
				employer = { 
					has_land = yes
				}
				is_alive = yes
				is_adult = yes
				NOT = {	has_trait = lower_vampire }
				NOT = {	has_trait = half_vampire }
				NOT = {	has_trait = full_vampire }
				NOT = {	has_trait = werewolf }
			}
			save_scope_as = candidate_gov
		}
		trigger_event = {
			id = vampirism.1
		}
	}
	
}

vampirism.4 = {
	type = country_event
	hidden = yes
	
	trigger = {
		religion = night_pantheon
	}
	
	immediate = {
		every_character = {
			limit = {
				is_alive = yes
				OR = {
					exists = father
					exists = mother
				}
			}
			if = {
				limit = {
					OR = {
						father = {
							has_trait = full_vampire
							is_ruler = yes
						}
						mother = {
							has_trait = full_vampire
							is_ruler = yes
						}
					}
					NOT = {	has_trait = full_vampire }
				}
				save_scope_as = childofnight
			}
		}
		trigger_event = {
			id = vampirism.5
		}
	}
}
