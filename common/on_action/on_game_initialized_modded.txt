	effect = {
		basic_unit_setup = yes

			value = oratory_tech
		}

		}

					has_global_variable = province_init_done
				}
			}

						is_sea = yes

					}
				}

			}
			set_global_variable = {
				name = province_init_done
				value = 1
			}

				limit = {

				}
